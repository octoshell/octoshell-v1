def breadcrumbs
  breadcrumbs = [self]
  
  if ticket_question
    breadcrumbs << ticket_question
    
    current_question = ticket_question
    while ticket_question = current_question.ticket_question
      current_question = ticket_question
      breadcrumbs << current_question
    end
  end
  
  breadcrumbs.reverse
end
