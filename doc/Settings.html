def self.update(settings)
  File.open("#{Rails.root}/config/settings.yml", "w+") do |file|
    file.write({ settings: settings.to_hash }.to_yaml)
  end
  @@settings = YAML.load_file("#{Rails.root}/config/settings.yml")[:settings]
end
